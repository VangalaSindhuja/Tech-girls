let shareCount = 0;
const shareBtn = document.getElementById("shrbtn");
const counterDisplay = document.getElementById("counter");
const finalMsg = document.getElementById("finalmsg");
const submitBtn = document.getElementById("sutbutn");
const form = document.getElementById("registration");

// Check if form already submitted
if (localStorage.getItem("formSubmitted") === "true") {
  disableForm();
  finalMsg.textContent = "ğŸ‰ Your submission has already been recorded. Thank you!";
}

// Share button logic
shareBtn.addEventListener("click", () => {
  if (shareCount < 5) {
    shareCount++;
    counterDisplay.textContent = `Click Count: ${shareCount}/5`;

    const message = encodeURIComponent("Hey buddy, Join Tech For Girls Community!");
    const whatsappURL = `https://wa.me/?text=${message}`;
    window.open(whatsappURL, "_blank");

    if (shareCount === 5) {
      finalMsg.textContent = "âœ… Sharing complete. Please continue.";
    }
  } else {
    finalMsg.textContent = "âœ… You already completed sharing.";
  }
});

// Submit button logic
submitBtn.addEventListener("click", () => {
  if (shareCount < 5) {
    alert("Please complete WhatsApp sharing first (5/5).");
    return;
  }

  // Get field values
  const name = document.getElementById("name").value.trim();
  const phone = document.getElementById("phone").value.trim();
  const email = document.getElementById("email").value.trim();
  const college = document.getElementById("college").value.trim();
  const file = document.getElementById("screenshot").files[0];

  // Validate each field
  if (!name) {
    alert("Please enter your Name.");
    document.getElementById("name").focus();
    return;
  }

  if (!phone) {
    alert("Please enter your Phone Number.");
    document.getElementById("phone").focus();
    return;
  }

  if (!email) {
    alert("Please enter your Email.");
    document.getElementById("email").focus();
    return;
  }

  if (!college) {
    alert("Please enter your College/Department.");
    document.getElementById("college").focus();
    return;
  }

  if (!file) {
    alert("Please upload your screenshot/file.");
    document.getElementById("screenshot").focus();
    return;
  }

  // All validations passed
  disableForm();
  localStorage.setItem("formSubmitted", "true");
  finalMsg.textContent = "ğŸ‰ Your submission has been recorded. Thanks for being part of Tech for Girls!";
});

// Disable form and buttons
function disableForm() {
  const allInputs = form.querySelectorAll("input");
  allInputs.forEach(input => input.disabled = true);
  shareBtn.disabled = true;
  submitBtn.disabled = true;
}
